```python
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        raise NotImplementedError("Subclasses must implement this method")

class Cat(Animal):
    def speak(self):
        return f"{self.name} says meow!"

# 创建Cat类的实例
my_cat = Cat("Whiskers")
print(my_cat.speak())  # 输出: Whiskers says meow!

```
